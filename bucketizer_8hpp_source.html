69  block allocate(size_t n) noexcept
70  {
71  size_t i = 0;

101  bool reallocate(block &b, size_t n) noexcept
102  {
103  if (n != 0 && (n < MinSize || n > MaxSize)) {
104  return false;
105  }
106 
107  if (internal::reallocator<bucketizer>::is_handled_default(*this, b, n)) {
108  return true;

110 
111  assert(owns(b));

113  const auto alignedLength = internal::round_to_alignment(StepSize, n);
114  auto currentAllocator = find_matching_allocator(b.length);
115  auto newAllocator = find_matching_allocator(alignedLength);
116 
117  if (currentAllocator == newAllocator) {
