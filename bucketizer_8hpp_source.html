72  block allocate(size_t n) {
73  size_t i = 0;
74  while (i < number_of_buckets) {

102  bool reallocate(block &b, size_t n) {
103  if (n != 0 && (n < MinSize || n > MaxSize)) {
104  return false;
105  }
106 
107  if (internal::reallocator<bucketizer>::isHandledDefault(*this, b, n)) {
108  return true;
109  }
110 
111  BOOST_ASSERT(owns(b));
112 
113  const auto alignedLength = internal::roundToAlignment(StepSize, n);
114  auto currentAllocator = findMatchingAllocator(b.length);
115  auto newAllocator = findMatchingAllocator(alignedLength);
116 
117  if (currentAllocator == newAllocator) {
118  return true;

157  BOOST_ASSERT(MinSize <= n && n <= MaxSize);
