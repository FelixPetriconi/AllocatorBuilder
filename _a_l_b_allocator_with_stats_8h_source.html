287  bool reallocate(Block& b, size_t n) {
288  auto originalBlock = b;
289  up(StatsOptions::numReallocate, _numReallocate);
290 
291  if (!_allocator.reallocate(b, n))
292  {
293  return false;
294  }
295  up(StatsOptions::numReallocateOK, _numReallocateOK);
296  std::make_signed<size_t>::type delta = b.length - originalBlock.length;
297  if (b.ptr == originalBlock.ptr) {
298  up(StatsOptions::numReallocateInPlace, _numReallocateInPlace);
299  if (delta > 0) {
300  add(StatsOptions::bytesAllocated, _bytesAllocated, delta);
301  add(StatsOptions::bytesExpanded, _bytesExpanded, delta);
302  }
303  else {
304  add(StatsOptions::bytesDeallocated, _bytesDeallocated, -delta);
305  add(StatsOptions::bytesContracted, _bytesContracted, -delta);
306  }
307  } // was moved to a new location
308  else {
309  add(StatsOptions::bytesAllocated, _bytesAllocated, b.length);
310  add(StatsOptions::bytesMoved, _bytesMoved, originalBlock.length);
311  add(StatsOptions::bytesDeallocated, _bytesDeallocated, originalBlock.length);
312  }
313  updateHighTide();
314  return true;
315  }
