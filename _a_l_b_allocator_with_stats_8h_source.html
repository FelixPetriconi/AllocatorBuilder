332  Block allocate(size_t n, const char *file = nullptr,
333  const char *function = nullptr, int line = 0) {
334  auto result = _allocator.allocate(n);
335  up(StatsOptions::NumAllocate, _numAllocate);
336  upOK(StatsOptions::NumAllocateOK, _numAllocateOK, n > 0 && result);
337  add(StatsOptions::BytesAllocated, _bytesAllocated, result.length);
338  updateHighTide();
339 
340  if (result && has_per_allocation_state) {
341  AllocationInfo *stat =
342  Traits::AffixExtractor<decltype(_allocator), AllocationInfo>::prefix(
343  _allocator, result);
344  set(StatsOptions::CallerSize, stat->callerSize, n);
345  set(StatsOptions::CallerFile, stat->callerFile, file);
346  set(StatsOptions::CallerFunction, stat->callerFunction, function);
347  set(StatsOptions::CallerLine, stat->callerLine, line);
348  set(StatsOptions::CallerTime, stat->callerTime,
349  std::chrono::system_clock::now());
350  // push into caller info stack
351  if (_root) {
352  _root->previous = stat;
353  stat->next = _root;
354  stat->previous = nullptr;
355  _root = stat;
356  } else { // create the new stack
357  stat->previous = nullptr;
358  stat->next = nullptr;
359  _root = stat;
360  }
361  }
362  return result;
363  }

400  bool reallocate(Block &b, size_t n) {
401  auto originalBlock = b;
402  bool wasRootBlock(false);
403  if (b && has_per_allocation_state) {
404  wasRootBlock = _root == Traits::AffixExtractor<
405  decltype(_allocator), AllocationInfo>::prefix(_allocator, b);
406  }
407  up(StatsOptions::NumReallocate, _numReallocate);
408 
409  if (!_allocator.reallocate(b, n)) {
410  return false;
411  }
412  up(StatsOptions::NumReallocateOK, _numReallocateOK);
413  std::make_signed<size_t>::type delta = b.length - originalBlock.length;
414  if (b.ptr == originalBlock.ptr) {
415  up(StatsOptions::NumReallocateInPlace, _numReallocateInPlace);
416  if (delta > 0) {
417  add(StatsOptions::BytesAllocated, _bytesAllocated, delta);
418  add(StatsOptions::BytesExpanded, _bytesExpanded, delta);
419  } else {
420  add(StatsOptions::BytesDeallocated, _bytesDeallocated, -delta);
421  add(StatsOptions::BytesContracted, _bytesContracted, -delta);
422  }
423  } // was moved to a new location
424  else {
425  add(StatsOptions::BytesAllocated, _bytesAllocated, b.length);
426  add(StatsOptions::BytesMoved, _bytesMoved, originalBlock.length);
427  add(StatsOptions::BytesDeallocated, _bytesDeallocated,
428  originalBlock.length);
429 
430  if (b && has_per_allocation_state) {
431  auto stat = Traits::AffixExtractor<
432  decltype(_allocator), AllocationInfo>::prefix(_allocator, b);
433  if (stat->next) {
434  stat->next->previous = stat;
435  }
436  if (stat->previous) {
437  stat->previous->next = stat;
438  }
439  if (wasRootBlock) {
440  _root = stat;
441  }
442  }
443  }
444  updateHighTide();
445  return true;
446  }
