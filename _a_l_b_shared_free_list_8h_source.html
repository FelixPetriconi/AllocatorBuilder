74  Block allocate(size_t n) {
75  BOOST_ASSERT_MSG(_lowerBound.value() != -1,
76  "The lower bound was not initialized!");
77  BOOST_ASSERT_MSG(_upperBound.value() != -1,
78  "The upper bound was not initialized!");
79 
80  if (_lowerBound.value() <= n && n <= _upperBound.value() && !_root.empty()) {
81  void* freeBlock = nullptr;
82 
83  if (_root.pop(freeBlock)) {
84  return Block(freeBlock, _upperBound.value() - 1);
85  }
86  else {
87  // allocating in a bunch to gain of having the allocator code in the cache
88  size_t blockSize = _upperBound.value();
89  auto batchAllocatedBlocks = _allocator.allocate(blockSize * NumberOfBatchAllocations);
90  if (batchAllocatedBlocks) {
91  // we use the very first block directly so we start at 1
92  for (size_t i = 1; i < NumberOfBatchAllocations; i++) {
93  if (!_root.push(static_cast<char*>(batchAllocatedBlocks.ptr) + i * blockSize)) {
94  BOOST_ASSERT(false);
95  _allocator.deallocate(Block(static_cast<char*>(batchAllocatedBlocks.ptr) + i * blockSize, blockSize));
96  }
97  }
98  // returning the first within block
99  return Block(batchAllocatedBlocks.ptr, blockSize);
100  }
101  return _allocator.allocate(blockSize);
102  }
103  }
104  return _allocator.allocate(static_cast<size_t>(_upperBound.value()));
105  }
