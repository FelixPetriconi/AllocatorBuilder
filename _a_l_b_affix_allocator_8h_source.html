58 template <class Allocator, typename Prefix,
59  typename Sufix = AffixAllocatorHelper::NoAffix>
60 class AffixAllocator
61 #ifdef BOOST_NO_CXX11_DELETED_FUNCTIONS
62  : public boost::noncopyable
63 #endif
64 {
65  Allocator _allocator;
66 
67  Prefix *innerToPrefix(const Block &b) const {
68  return reinterpret_cast<Prefix *>(static_cast<Prefix *>(b.ptr));
69  }
70 
71  Sufix *innerToSufix(const Block &b) const {
72  return reinterpret_cast<Sufix *>(static_cast<char *>(b.ptr) + b.length -
73  sufix_size);
74  }
75 
76  Block toInnerBlock(const Block &b) const {
77  return Block(static_cast<char *>(b.ptr) - prefix_size,
78  b.length + prefix_size + sufix_size);
79  }
80 
81  Block toOuterBlock(const Block &b) const {
82  return Block(static_cast<char *>(b.ptr) + prefix_size,
83  b.length - prefix_size - sufix_size);
84  }

87  Allocator::supports_truncated_deallocation;

96  : sizeof(Prefix);

144  Block allocate(size_t n) {
145  if (n == 0) {
146  return Block();
147  }
148 
149  auto innerMem = _allocator.allocate(prefix_size + n + sufix_size);
150  if (innerMem) {
151  if (prefix_size > 0) {
152  new (innerToPrefix(innerMem)) Prefix();
153  }
154  if (sufix_size > 0) {
155  new (innerToSufix(innerMem)) Sufix();
156  }
157  return toOuterBlock(innerMem);
158  }
159  return Block();
160  }
161 
168  void deallocate(Block &b) {
169  if (!b) {
170  return;
171  }
172  if (prefix_size > 0) {
173  outerToPrefix(b)->~Prefix();
174  }
175  if (sufix_size > 0) {
176  outerToSufix(b)->~Sufix();
177  }
178  _allocator.deallocate(toInnerBlock(b));
179  b.reset();
180  }

187  typename Traits::enable_result_to<bool,
188  Traits::has_owns<Allocator>::value>::type
189  owns(const Block &b) const {
190  return b && _allocator.owns(toInnerBlock(b));
191  }

200  bool reallocate(Block &b, size_t n) {
201  if (Helper::Reallocator<AffixAllocator>::isHandledDefault(*this, b, n)) {
202  return true;
203  }
204  auto innerBlock = toInnerBlock(b);
205 
206  // Remember the old Sufix in case that it is available, because it must
207  // be later placed to the new position
208  boost::optional<Sufix> oldSufix;
209  if (sufix_size > 0) {
210  oldSufix.reset(*outerToSufix(b));
211  }
212 
213  if (_allocator.reallocate(innerBlock, n + prefix_size + sufix_size)) {
214  if (sufix_size > 0) {
215  new (innerToSufix(innerBlock)) Sufix(*oldSufix);
216  }
217  b = toOuterBlock(innerBlock);
218  return true;
219  }
220  return false;
221  }

231  typename Traits::enable_result_to<bool,
232  Traits::has_expand<Allocator>::value>::type
233  expand(Block &b, size_t delta) {
234  if (delta == 0) {
235  return true;
236  }
237 
238  if (!b) {
239  b = allocate(delta);
240  return static_cast<bool>(b);
241  }
242 
243  auto oldBlock = b;
244  auto innerBlock = toInnerBlock(b);
245 
246  if (_allocator.expand(innerBlock, delta)) {
247  if (sufix_size > 0) {
248  new (innerToSufix(innerBlock)) Sufix(*outerToSufix(oldBlock));
249  }
250  b = toOuterBlock(innerBlock);
251  return true;
252  }
253  return false;
254  }
