70  Block allocateNoGrow(size_t n) {
71  Block result;
72  Node *p = _root.load();
73  while (p) {
74  result = p->allocator.allocate(n);
75  if (result) {
76  return result;
77  }

86  Node* createNode() {
87  // Create a temporary node with an allocator on the stack
88  Node nodeOnStack;
89 
90  // Use this allocator to create the first node in allocators space
91  auto nodeBlock = nodeOnStack.allocator.allocate(sizeof(Node));
92 
93  nodeOnStack.allocatedThisSize = nodeBlock.length;
94  Node* result = reinterpret_cast<Node*>(nodeBlock.ptr);
95 
96  if (!result) {
97  return nullptr;
98  }
99

101  new (result) Node();
102 
103  // Move the node from the stack to the allocated space
104  *result = std::move(nodeOnStack);

112  void eraseNode(Node* n) {
113  if (n == nullptr) {
114  return;
115  }
116  if (n->next.load()) {
117  // delete all possible next Nodes in the list
118  eraseNode(n->next.load());

184  Block allocate(size_t n) {
185  Block result = allocateNoGrow(n);
186  if (result) {
187  return result;
188  }
189 
190  // no node at all there
191  if (_root.load() == nullptr) {
192  Node* firstNode = createNode();
193  Node* nullNode = nullptr;
194  // test if in the meantime someone else has created a node
195  if (!_root.compare_exchange_strong(nullNode, firstNode)) {
196  eraseNode(firstNode);
197  }
198 
199  result = allocateNoGrow(n);
200  if (result) {
201  return result;
202  }
203  }
204 
205  // a new node must be appended
206  Node *newNode = createNode();
207  Node *nullNode = nullptr;
208  Node *p;
209  do {
210  p = _root;
211  while (p->next.load() != nullptr) {
212  p = p->next;
213  }
214  } while (!p->next.compare_exchange_strong(nullNode, newNode));
215 
216  result = allocateNoGrow(n);
217  return result;
218  }
