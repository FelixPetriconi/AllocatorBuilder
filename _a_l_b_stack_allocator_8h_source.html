26  template <size_t MaxSize, size_t Alignment = 4>
27  class StackAllocator {
28  char _data[MaxSize];
29  char* _p;
30 
31  bool isLastUsedBlock(const Block& block) const {
32  return (static_cast<char*>(block.ptr) + block.length == _p);
33  }
34 
35  public:
36  typename typedef StackAllocator allocator;
37  static const size_t max_size = MaxSize;
38  static const size_t alignment = Alignment;
39 
40  StackAllocator() : _p(_data) {}

42  Block allocate(size_t n) {
43  Block result;
44 
45  if (n == 0) {
46  return result;
47  }
48 
49  const auto alignedLength = Helper::roundToAlignment(Alignment, n);
50  if (alignedLength + _p > _data + MaxSize) { // not enough memory left
51  return result;
52  }
53 
54  result.ptr = _p;
55  result.length = alignedLength;
56 
57  _p += alignedLength;
58  return result;
59  }
60 
61  void deallocate(Block& b) {
62  if (!b) {
63  return;
64  }
65  if (!owns(b)) {
66  BOOST_ASSERT(false);
67  return;
68  }
69 
70  // If it was the most recent allocated MemoryBlock, then we can re-use the memory. Otherwise
71  // this freed MemoryBlock is not available for further allocations. Since all happens on the stack
72  // this is not a leak!
73  if (isLastUsedBlock(b)) {
74  _p = static_cast<char*>(b.ptr);
75  }
76  b.reset();
77  }
78 
79  bool reallocate(Block& b, size_t n) {
80  if (b.length == n) {
81  return true;
82  }
83 
84  if (n == 0) {
85  deallocate(b);
86  return true;
87  }
88 
89  if (!b) {
90  b = allocate(n);
91  return true;
92  }
93 
94  const auto alignedLength = Helper::roundToAlignment(Alignment, n);
95 
96  if ( isLastUsedBlock(b) ) {
97  if (static_cast<char*>(b.ptr) + alignedLength <= _data + MaxSize) {
98  b.length = alignedLength;
99  _p = static_cast<char*>(b.ptr) + alignedLength;
100  return true;
101  }
102  else { // out of memory
103  return false;
104  }
105  }
106  else {
107  if (b.length > n) {
108  b.length = Helper::roundToAlignment(Alignment, n);
109  return true;
110  }
111  }
112 
113  auto newBlock = allocate(alignedLength);
114  // we cannot deallocate the old block, because it is in between used ones, so we have to
115  // "leak" here.
116  if (newBlock) {
117  Helper::blockCopy(b, newBlock);
118  b = newBlock;
119  return true;
120  }
121  return false;
122  }
123 
131  bool expand(Block& b, size_t delta) {
132  if (delta == 0) {
133  return true;
134  }
135  if (!b) {
136  b = allocate(delta);
137  return b.length != 0;
138  }
139  if (!isLastUsedBlock(b)) {
140  return false;
141  }
142  auto alignedBytes = Helper::roundToAlignment(Alignment, delta);
143  if (_p + alignedBytes > _data + MaxSize) {
144  return false;
145  }
146  _p += alignedBytes;
147  b.length += alignedBytes;
148  return true;
149  }

156  bool owns(const Block& b) const {
157  return b && (b.ptr >= _data && b.ptr < _data + MaxSize);
158  }
