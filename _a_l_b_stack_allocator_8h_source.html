30 template <size_t MaxSize, size_t Alignment = 4>
31 class StackAllocator
32 #ifdef BOOST_NO_CXX11_DELETED_FUNCTIONS
33  : boost::noncopyable
34 #endif
35 {
36  char _data[MaxSize];
37  char *_p;
38 
39  bool isLastUsedBlock(const Block &block) const {
40  return (static_cast<char *>(block.ptr) + block.length == _p);
41  }
42 
43 public:
44  static const bool supports_truncated_deallocation = true;
45  typename typedef StackAllocator allocator;
46  static const size_t max_size = MaxSize;
47  static const size_t alignment = Alignment;
48 
49  StackAllocator() : _p(_data) {}

51  Block allocate(size_t n) {
52  Block result;
53 
54  if (n == 0) {
55  return result;
56  }
57 
58  const auto alignedLength = Helper::roundToAlignment(Alignment, n);
59  if (alignedLength + _p > _data + MaxSize) { // not enough memory left
60  return result;
61  }
62 
63  result.ptr = _p;
64  result.length = alignedLength;
65 
66  _p += alignedLength;
67  return result;
68  }
69 
70  void deallocate(Block &b) {
71  if (!b) {
72  return;
73  }
74  if (!owns(b)) {
75  BOOST_ASSERT(false);
76  return;
77  }
78 
79  // If it was the most recent allocated MemoryBlock, then we can re-use the
80  // memory. Otherwise this freed MemoryBlock is not available for further
81  // allocations. Since all happens on the stack this is not a leak!
82  if (isLastUsedBlock(b)) {
83  _p = static_cast<char *>(b.ptr);
84  }
85  b.reset();
86  }
87 
88  bool reallocate(Block &b, size_t n) {
89  if (b.length == n) {
90  return true;
91  }
92 
93  if (n == 0) {
94  deallocate(b);
95  return true;
96  }
97 
98  if (!b) {
99  b = allocate(n);
100  return true;
101  }
102 
103  const auto alignedLength = Helper::roundToAlignment(Alignment, n);
104 
105  if (isLastUsedBlock(b)) {
106  if (static_cast<char *>(b.ptr) + alignedLength <= _data + MaxSize) {
107  b.length = alignedLength;
108  _p = static_cast<char *>(b.ptr) + alignedLength;
109  return true;
110  } else { // out of memory
111  return false;
112  }
113  } else {
114  if (b.length > n) {
115  b.length = Helper::roundToAlignment(Alignment, n);
116  return true;
117  }
118  }
119 
120  auto newBlock = allocate(alignedLength);
121  // we cannot deallocate the old block, because it is in between used ones,
122  // so we have to "leak" here.
123  if (newBlock) {
124  Helper::blockCopy(b, newBlock);
125  b = newBlock;
126  return true;
127  }
128  return false;
129  }
130 
138  bool expand(Block &b, size_t delta) {
139  if (delta == 0) {
140  return true;
141  }
142  if (!b) {
143  b = allocate(delta);
144  return b.length != 0;
145  }
146  if (!isLastUsedBlock(b)) {
147  return false;
148  }
149  auto alignedBytes = Helper::roundToAlignment(Alignment, delta);
150  if (_p + alignedBytes > _data + MaxSize) {
151  return false;
152  }
153  _p += alignedBytes;
154  b.length += alignedBytes;
155  return true;
156  }
157 
163  bool owns(const Block &b) const {
164  return b && (b.ptr >= _data && b.ptr < _data + MaxSize);
165  }
166 
172  void deallocateAll() { _p = _data; }

179 #endif
