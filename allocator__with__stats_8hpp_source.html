333  block allocate(size_t n, const char *file = nullptr,
334  const char *function = nullptr, int line = 0) {
335  auto result = _allocator.allocate(n);
336  up(StatsOptions::NumAllocate, _numAllocate);
337  upOK(StatsOptions::NumAllocateOK, _numAllocateOK, n > 0 && result);
338  add(StatsOptions::BytesAllocated, _bytesAllocated, result.length);
339  updateHighTide();
340 
341  if (result && has_per_allocation_state) {
342  AllocationInfo *stat =
343  traits::affix_extractor<decltype(_allocator), AllocationInfo>::prefix(
344  _allocator, result);
345 
346  set(StatsOptions::CallerSize, stat->callerSize, n);
347  set(StatsOptions::CallerFile, stat->callerFile, file);
348  set(StatsOptions::CallerFunction, stat->callerFunction, function);
349  set(StatsOptions::CallerLine, stat->callerLine, line);
350  set(StatsOptions::CallerTime, stat->callerTime,
351  std::chrono::system_clock::now());
352 
353  // push into caller info stack
354  if (_root) {
355  _root->previous = stat;
356  stat->next = _root;
357  stat->previous = nullptr;
358  _root = stat;
359  } else { // create the new stack
360  stat->previous = nullptr;
361  stat->next = nullptr;
362  _root = stat;
363  }
364  }
365  return result;
366  }

403  bool reallocate(block &b, size_t n) {
404  auto originalBlock = b;
405  bool wasRootBlock(false);
406  if (b && has_per_allocation_state) {
407  wasRootBlock = _root == traits::affix_extractor<
408  decltype(_allocator), AllocationInfo>::prefix(_allocator, b);
409  }
410  up(StatsOptions::NumReallocate, _numReallocate);
411 
412  if (!_allocator.reallocate(b, n)) {
413  return false;
414  }
415  up(StatsOptions::NumReallocateOK, _numReallocateOK);
416  std::make_signed<size_t>::type delta = b.length - originalBlock.length;
417  if (b.ptr == originalBlock.ptr) {
418  up(StatsOptions::NumReallocateInPlace, _numReallocateInPlace);
419  if (delta > 0) {
420  add(StatsOptions::BytesAllocated, _bytesAllocated, delta);
421  add(StatsOptions::BytesExpanded, _bytesExpanded, delta);
422  } else {
423  add(StatsOptions::BytesDeallocated, _bytesDeallocated, -delta);
424  add(StatsOptions::BytesContracted, _bytesContracted, -delta);
425  }
426  } // was moved to a new location
427  else {
428  add(StatsOptions::BytesAllocated, _bytesAllocated, b.length);
429  add(StatsOptions::BytesMoved, _bytesMoved, originalBlock.length);
430  add(StatsOptions::BytesDeallocated, _bytesDeallocated,
431  originalBlock.length);
432 
433  if (b && has_per_allocation_state) {
434  auto stat = traits::affix_extractor<
435  decltype(_allocator), AllocationInfo>::prefix(_allocator, b);
436  if (stat->next) {
437  stat->next->previous = stat;
438  }
439  if (stat->previous) {
440  stat->previous->next = stat;
441  }
442  if (wasRootBlock) {
443  _root = stat;
444  }
445  }
446  }
447  updateHighTide();
448  return true;
449  }
