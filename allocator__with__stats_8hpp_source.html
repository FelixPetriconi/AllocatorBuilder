413  block allocate(size_t n, const char *file = nullptr, const char *function = nullptr,
414  int line = 0)
415  {
416 
417  auto result = _allocator.allocate(n);
418  up(StatsOptions::NumAllocate, _numAllocate);
419  upOK(StatsOptions::NumAllocateOK, _numAllocateOK, n > 0 && result);
420  add(StatsOptions::BytesAllocated, _bytesAllocated, result.length);
421  updateHighTide();
422 
423  if (result && has_per_allocation_state) {
424  AllocationInfo *stat =
425  traits::affix_extractor<decltype(_allocator), AllocationInfo>::prefix(_allocator,
426  result);
427 
428  set(StatsOptions::CallerSize, stat->callerSize, n);
429  set(StatsOptions::CallerFile, stat->callerFile, file);
430  set(StatsOptions::CallerFunction, stat->callerFunction, function);
431  set(StatsOptions::CallerLine, stat->callerLine, line);
432  set(StatsOptions::CallerTime, stat->callerTime, std::chrono::system_clock::now());
433 
434  // push into caller info stack
435  if (_root) {
436  _root->previous = stat;
437  stat->next = _root;
438  stat->previous = nullptr;
439  _root = stat;
440  }
441  else { // create the new stack
442  stat->previous = nullptr;
443  stat->next = nullptr;
444  _root = stat;
445  }
446  }
447  return result;
448  }

456  void deallocate(block &b)
457  {
458  up(StatsOptions::NumDeallocate, _numDeallocate);
459  add(StatsOptions::BytesDeallocated, _bytesDeallocated, b.length);
460 
461  if (b && has_per_allocation_state) {

485  bool reallocate(block &b, size_t n)
486  {
487  auto originalBlock = b;
488  auto wasRootBlock(false);
489  if (b && has_per_allocation_state) {
490  wasRootBlock =
491  _root ==
492  traits::affix_extractor<decltype(_allocator), AllocationInfo>::prefix(_allocator, b);
493  }
494  up(StatsOptions::NumReallocate, _numReallocate);
495 
496  if (!_allocator.reallocate(b, n)) {
497  return false;
498  }
499  up(StatsOptions::NumReallocateOK, _numReallocateOK);
500  std::make_signed<size_t>::type delta = b.length - originalBlock.length;
501  if (b.ptr == originalBlock.ptr) {
502  up(StatsOptions::NumReallocateInPlace, _numReallocateInPlace);
503  if (delta > 0) {
504  add(StatsOptions::BytesAllocated, _bytesAllocated, delta);
505  add(StatsOptions::BytesExpanded, _bytesExpanded, delta);
506  }
507  else {
508  add(StatsOptions::BytesDeallocated, _bytesDeallocated, -delta);
509  add(StatsOptions::BytesContracted, _bytesContracted, -delta);
510  }
511  } // was moved to a new location
512  else {
513  add(StatsOptions::BytesAllocated, _bytesAllocated, b.length);
514  add(StatsOptions::BytesMoved, _bytesMoved, originalBlock.length);
515  add(StatsOptions::BytesDeallocated, _bytesDeallocated, originalBlock.length);
516 
517  if (b && has_per_allocation_state) {
518  auto stat =
519  traits::affix_extractor<decltype(_allocator), AllocationInfo>::prefix(_allocator, b);
520  if (stat->next) {
521  stat->next->previous = stat;
522  }
523  if (stat->previous) {
524  stat->previous->next = stat;
525  }
526  if (wasRootBlock) {
527  _root = stat;
528  }
529  }
530  }
531  updateHighTide();
532  return true;
533  }
