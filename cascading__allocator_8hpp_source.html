72  block allocateNoGrow(size_t n) {
73  block result;
74  Node *p = _root.load();
75  while (p) {
76  result = p->allocator.allocate(n);
77  if (result) {

88  Node* createNode() {
89  // Create a temporary node with an allocator on the stack
90  Node nodeOnStack;
91 
92  // Use this allocator to create the first node in allocators space
93  auto nodeBlock = nodeOnStack.allocator.allocate(sizeof(Node));
94 
95  nodeOnStack.allocatedThisSize = nodeBlock.length;
96  Node* result = reinterpret_cast<Node*>(nodeBlock.ptr);
97 
98  if (!result) {
99  return nullptr;
100  }
101 
102  // Create a new Node emplace
103  new (result) Node();
104 
105  // Move the node from the stack to the allocated space
106  *result = std::move(nodeOnStack);
107 
108  return result;
109  }

114  void eraseNode(Node* n) {
115  if (n == nullptr) {
116  return;
117  }
118  if (n->next.load()) {
119  // delete all possible next Nodes in the list
120  eraseNode(n->next.load());
121  n->next = nullptr;
122  }
123  // Create a temporary node on the stack
124  Node stackNode;
125 
126  // Move the allocator to the temporary node
127  stackNode = std::move(*n);
128  block allocatedBlock(n, stackNode.allocatedThisSize);
129 
130  stackNode.allocator.deallocate(allocatedBlock);
131  }

186  block allocate(size_t n) {
187  block result = allocateNoGrow(n);
188  if (result) {
189  return result;
190  }
191 
192  // no node at all there
193  if (_root.load() == nullptr) {
194  Node* firstNode = createNode();
195  Node* nullNode = nullptr;
196  // test if in the meantime someone else has created a node
197  if (!_root.compare_exchange_strong(nullNode, firstNode)) {
198  eraseNode(firstNode);
199  }
200 
201  result = allocateNoGrow(n);
202  if (result) {
203  return result;
204  }
205  }
206 
207  // a new node must be appended
208  Node *newNode = createNode();
209  Node *nullNode = nullptr;
210  Node *p;
211  do {
212  p = _root;
213  while (p->next.load() != nullptr) {
214  p = p->next;
215  }
216  } while (!p->next.compare_exchange_strong(nullNode, newNode));
217 
218  result = allocateNoGrow(n);
219  return result;
220  }
