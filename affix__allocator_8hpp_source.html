164  block allocate(size_t n) noexcept
165  {
166  if (n == 0) {
167  return{};
168  }
169 
170  auto innerMem = allocator_.allocate(prefix_size + n + sufix_size);
171  if (innerMem) {
172  if (prefix_size > 0) {
173  new (inner_to_prefix(innerMem)) Prefix{};
174  }
175  if (sufix_size > 0) {
176  new (inner_to_sufix(innerMem)) Sufix{};
177  }
178  return to_outer_block(innerMem);
179  }
180  return{};
181  }
182 
188  void deallocate(block &b) noexcept
189  {
190  if (!b) {
191  return;
192  }
193  if (prefix_size > 0) {
194  outer_to_prefix(b)->~Prefix();
195  }
196  if (sufix_size > 0) {
197  outer_to_sufix(b)->~Sufix();
198  }
199  auto innerBlock(to_inner_block(b));
200  allocator_.deallocate(innerBlock);
201  b.reset();
202  }

224  bool reallocate(block &b, size_t n) noexcept
225  {
226  if (internal::reallocator<affix_allocator>::is_handled_default(*this, b, n)) {
227  return true;
228  }
229  auto innerBlock = to_inner_block(b);
230 
231  // Remember the old Sufix in case that it is available, because it must
232  // be later placed to the new position
233  affix_allocator_helper::optinal_sufix_store<Sufix, sufix_size> oldSufix;
234  oldSufix.store(outer_to_sufix(b));
235 
236  if (allocator_.reallocate(innerBlock, n + prefix_size + sufix_size)) {
237  oldSufix.unload(inner_to_sufix(innerBlock));
238  b = to_outer_block(innerBlock);
239  return true;
240  }
241  return false;
242  }

252  template <typename U = Allocator>
253  typename std::enable_if<traits::has_expand<U>::value, bool>::type
254  expand(block &b, size_t delta) noexcept
255  {
256  if (delta == 0) {
257  return true;
258  }
259 
260  if (!b) {
261  b = allocate(delta);
262  return static_cast<bool>(b);
263  }
264 
265  auto oldBlock = b;
266  auto innerBlock = to_inner_block(b);
267 
268  if (allocator_.expand(innerBlock, delta)) {
269  if (sufix_size > 0) {
270  new (inner_to_sufix(innerBlock)) Sufix(*outer_to_sufix(oldBlock));
271  }
272  b = to_outer_block(innerBlock);
273  return true;
274  }
275  return false;
276  }
