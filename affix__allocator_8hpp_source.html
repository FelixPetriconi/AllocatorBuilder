149  block allocate(size_t n) {
150  if (n == 0) {
151  return block();
152  }
153 
154  auto innerMem = _allocator.allocate(prefix_size + n + sufix_size);
155  if (innerMem) {
156  if (prefix_size > 0) {
157  new (innerToPrefix(innerMem)) Prefix();
158  }
159  if (sufix_size > 0) {
160  new (innerToSufix(innerMem)) Sufix();
161  }
162  return toOuterBlock(innerMem);
163  }
164  return block();
165  }

173  void deallocate(block &b) {
174  if (!b) {
175  return;
176  }
177  if (prefix_size > 0) {
178  outerToPrefix(b)->~Prefix();
179  }
180  if (sufix_size > 0) {
181  outerToSufix(b)->~Sufix();
182  }
183  auto innerBlock(toInnerBlock(b));
184  _allocator.deallocate(innerBlock);
185  b.reset();
186  }

206  bool reallocate(block &b, size_t n) {
207  if (internal::reallocator<affix_allocator>::isHandledDefault(*this, b, n)) {
208  return true;
209  }
210  auto innerBlock = toInnerBlock(b);
211 
212  // Remember the old Sufix in case that it is available, because it must
213  // be later placed to the new position
214  boost::optional<Sufix> oldSufix;
215  if (sufix_size > 0) {
216  oldSufix.reset(*outerToSufix(b));
217  }
218 
219  if (_allocator.reallocate(innerBlock, n + prefix_size + sufix_size)) {
220  if (sufix_size > 0) {
221  new (innerToSufix(innerBlock)) Sufix(*oldSufix);
222  }
223  b = toOuterBlock(innerBlock);
224  return true;
225  }
226  return false;
227  }

237  typename traits::enable_result_to<bool,
238  traits::has_expand<Allocator>::value>::type
239  expand(block &b, size_t delta) {
240  if (delta == 0) {
241  return true;
242  }
243 
244  if (!b) {
245  b = allocate(delta);
246  return static_cast<bool>(b);
247  }
248 
249  auto oldBlock = b;
250  auto innerBlock = toInnerBlock(b);
251 
252  if (_allocator.expand(innerBlock, delta)) {
253  if (sufix_size > 0) {
254  new (innerToSufix(innerBlock)) Sufix(*outerToSufix(oldBlock));
255  }
256  b = toOuterBlock(innerBlock);
257  return true;
258  }
259  return false;
