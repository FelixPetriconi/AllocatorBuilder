140  block allocate(size_t n)
141  {
142  if (n == 0) {
143  return {};
144  }
145 
146  auto innerMem = _allocator.allocate(prefix_size + n + sufix_size);
147  if (innerMem) {
148  if (prefix_size > 0) {
149  new (innerToPrefix(innerMem)) Prefix{};
150  }
151  if (sufix_size > 0) {
152  new (innerToSufix(innerMem)) Sufix{};
153  }
154  return toOuterBlock(innerMem);
155  }
156  return {};
157  }
158 
165  void deallocate(block &b)
166  {
167  if (!b) {
168  return;
169  }
170  if (prefix_size > 0) {
171  outerToPrefix(b)->~Prefix();
172  }
173  if (sufix_size > 0) {
174  outerToSufix(b)->~Sufix();
175  }
176  auto innerBlock(toInnerBlock(b));
177  _allocator.deallocate(innerBlock);
178  b.reset();
179  }

200  bool reallocate(block &b, size_t n)
201  {
202  if (internal::reallocator<affix_allocator>::isHandledDefault(*this, b, n)) {
203  return true;
204  }
205  auto innerBlock = toInnerBlock(b);
206 
207  // Remember the old Sufix in case that it is available, because it must
208  // be later placed to the new position
209  boost::optional<Sufix> oldSufix;
210  if (sufix_size > 0) {
211  oldSufix.reset(*outerToSufix(b));
212  }
213 
214  if (_allocator.reallocate(innerBlock, n + prefix_size + sufix_size)) {
215  if (sufix_size > 0) {
216  new (innerToSufix(innerBlock)) Sufix(*oldSufix);
217  }
218  b = toOuterBlock(innerBlock);
219  return true;
220  }
221  return false;
222  }

232  template <typename U = Allocator>
233  typename std::enable_if<traits::has_expand<U>::value, bool>::type
234  expand(block &b, size_t delta)
235  {
236  if (delta == 0) {
237  return true;
238  }
239 
240  if (!b) {
241  b = allocate(delta);
242  return static_cast<bool>(b);
243  }
244 
245  auto oldBlock = b;
246  auto innerBlock = toInnerBlock(b);
247 
248  if (_allocator.expand(innerBlock, delta)) {
249  if (sufix_size > 0) {
250  new (innerToSufix(innerBlock)) Sufix(*outerToSufix(oldBlock));
251  }
252  b = toOuterBlock(innerBlock);
253  return true;
254  }
255  return false;
