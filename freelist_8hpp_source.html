117  void *curBlock = nullptr;

135  block allocate(size_t n) {
136  BOOST_ASSERT_MSG(_lowerBound.value() != -1,
137  "The lower bound was not initialized!");
138  BOOST_ASSERT_MSG(_upperBound.value() != -1,
139  "The upper bound was not initialized!");
140 
141  if (_lowerBound.value() <= n && n <= _upperBound.value()) {
142  void *freeBlock = nullptr;
143 
144  if (_root.pop(freeBlock)) {
145  return block(freeBlock, _upperBound.value());
146  } else {
147  size_t blockSize = _upperBound.value();
148  if (supports_truncated_deallocation) {
149  // allocating in a bunch to gain of having the allocator code in the
150  // cache
151  auto batchAllocatedBlocks =
152  _allocator.allocate(blockSize * NumberOfBatchAllocations);
153 
154  if (batchAllocatedBlocks) {
155  // we use the very first block directly so we start at 1
156  for (size_t i = 1; i < NumberOfBatchAllocations; i++) {
157  if (!_root.push(static_cast<char *>(batchAllocatedBlocks.ptr) +
158  i * blockSize)) {
159  BOOST_ASSERT(false);
160  alb::block oldBlock(static_cast<char *>(batchAllocatedBlocks.ptr) +
161  i * blockSize,
162  blockSize);
163  _allocator.deallocate(oldBlock);
164  }
165  }
166  // returning the first within block
167  return block(batchAllocatedBlocks.ptr, blockSize);
168  }
169  return _allocator.allocate(blockSize);
170  } else {
171  for (size_t i = 0; i < NumberOfBatchAllocations - 1; i++) {
172  auto b = _allocator.allocate(blockSize);
173  if (!_root.push(b.ptr)) { // the list is full in the meantime, so we
174  // exit early
175  return b;
176  }
177  }
178  return _allocator.allocate(blockSize);
179  }
180  }
181  }
182  return block();
183  }
