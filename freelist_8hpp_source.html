130  block allocate(size_t n) {
131  BOOST_ASSERT_MSG(_lowerBound.value() != -1,
132  "The lower bound was not initialized!");
133  BOOST_ASSERT_MSG(_upperBound.value() != -1,
134  "The upper bound was not initialized!");
135 
136  if (_lowerBound.value() <= n && n <= _upperBound.value()) {
137  void *freeBlock = nullptr;
138 
139  if (_root.pop(freeBlock)) {
140  return block(freeBlock, _upperBound.value());
141  } else {
142  size_t blockSize = _upperBound.value();
143  if (supports_truncated_deallocation) {
144  // allocating in a bunch to gain of having the allocator code in the
145  // cache
146  auto batchAllocatedBlocks =
147  _allocator.allocate(blockSize * NumberOfBatchAllocations);
148 
149  if (batchAllocatedBlocks) {
150  // we use the very first block directly so we start at 1
151  for (size_t i = 1; i < NumberOfBatchAllocations; i++) {
152  if (!_root.push(static_cast<char *>(batchAllocatedBlocks.ptr) +
153  i * blockSize)) {
154  BOOST_ASSERT(false);
155  alb::block oldBlock(static_cast<char *>(batchAllocatedBlocks.ptr) +
156  i * blockSize,
157  blockSize);
158  _allocator.deallocate(oldBlock);
159  }
160  }
161  // returning the first within block
162  return block(batchAllocatedBlocks.ptr, blockSize);
163  }
164  return _allocator.allocate(blockSize);
165  } else {
166  for (size_t i = 0; i < NumberOfBatchAllocations - 1; i++) {
167  auto b = _allocator.allocate(blockSize);
168  if (!_root.push(b.ptr)) { // the list is full in the meantime, so we
169  // exit early
170  return b;
171  }
172  }
173  return _allocator.allocate(blockSize);
174  }
175  }
176  }
177  return block();
178  }
