124  Block allocate(size_t n) {
125  BOOST_ASSERT_MSG(_lowerBound.value() != -1,
126  "The lower bound was not initialized!");
127  BOOST_ASSERT_MSG(_upperBound.value() != -1,
128  "The upper bound was not initialized!");
129 
130  if (_lowerBound.value() <= n && n <= _upperBound.value()) {
131  void *freeBlock = nullptr;
132 
133  if (_root.pop(freeBlock)) {
134  return Block(freeBlock, _upperBound.value());
135  } else {
136  size_t blockSize = _upperBound.value();
137  if (supports_truncated_deallocation) {
138  // allocating in a bunch to gain of having the allocator code in the
139  // cache
140  auto batchAllocatedBlocks =
141  _allocator.allocate(blockSize * NumberOfBatchAllocations);
142 
143  if (batchAllocatedBlocks) {
144  // we use the very first block directly so we start at 1
145  for (size_t i = 1; i < NumberOfBatchAllocations; i++) {
146  if (!_root.push(static_cast<char *>(batchAllocatedBlocks.ptr) +
147  i * blockSize)) {
148  BOOST_ASSERT(false);
149  _allocator.deallocate(
150  Block(static_cast<char *>(batchAllocatedBlocks.ptr) +
151  i * blockSize,
152  blockSize));
153  }
154  }
155  // returning the first within block
156  return Block(batchAllocatedBlocks.ptr, blockSize);
157  }
158  return _allocator.allocate(blockSize);
159  } else {
160  for (size_t i = 0; i < NumberOfBatchAllocations - 1; i++) {
161  auto b = _allocator.allocate(blockSize);
162  if (!_root.push(b.ptr)) { // the list is full in the meantime, so we
163  // exit early
164  return b;
165  }
166  }
167  return _allocator.allocate(blockSize);
168  }
169  }
170  }
171  return Block();
172  }
